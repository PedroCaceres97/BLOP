359blopt_vector blopfn_vector_create() {
360 blopt_vector vec = BLOP_MALLOC(struct blops_vector, 1);
361 if (!vec) {
362 BLOP_ERROR_MESSAGE("Failed to allocate memory for vector structure");
363 BLOP_ABORT();
364 return NULL;
365 }
366
367 vec->data = BLOP_MALLOC(BLOP_VECTOR_DATA_TYPE, BLOP_VECTOR_INITIAL_COUNT);
368 if (!vec->data) {
369 BLOP_ERROR_MESSAGE("Failed to allocate memory for vector->data");
370 free(vec);
371 BLOP_ABORT();
372 return NULL;
373 }
374
375 memset(vec->data, 0, BLOP_VECTOR_INITIAL_COUNT * sizeof(BLOP_VECTOR_DATA_TYPE));
376 vec->capacity = BLOP_VECTOR_INITIAL_COUNT;
377 vec->count = 0;
378 return vec;
379}

404void blopfn_vector_insert(blopt_vector vec, size_t idx, BLOP_VECTOR_DATA_TYPE value) {
405 BLOP_ASSERT_PTR_VOID(vec);
406
407 BLOP_ASSERT_BONDS_VOID(idx, vec->count);
408
409 if (idx == 0) {
410 blopfn_vector_push_front(vec, value);
411 return;
412 }
413
414 if (idx == vec->count - 1) {
415 blopfn_vector_push_back(vec, value);
416 return;
417 }
418
419 if (vec->count == vec->capacity) {
420 size_t capacity = vec->capacity * BLOP_VECTOR_SCALATOR;
421 BLOP_VECTOR_DATA_TYPE* data = BLOP_MALLOC(BLOP_VECTOR_DATA_TYPE, capacity);
422 if (!data) {
423 BLOP_ERROR_MESSAGE("Failed to allocate memory for vector data");
424 BLOP_ABORT();
425 return;
426 }
427
428 /*
429 The new element is inserted at index idx, so we need to copy:
430 - from 0 to idx-1, as idx element is the idx + 1 number of element
431 we only need to copy idx elements (0 to idx-1 contains exactly idx number of elements)
432 - then we place the new element at index idx
433 - we copy from idx to count-1, which is (count - idx) elements
434 - finally we fill with zeros the remaining capacity
435 */
436 memcpy(data, vec->data, idx * sizeof(BLOP_VECTOR_DATA_TYPE));
437 data[idx] = value;
438 memcpy(&data[idx + 1], &vec->data[idx], (vec->count - idx) * sizeof(BLOP_VECTOR_DATA_TYPE));
439 memset(&data[vec->count + 1], 0, (capacity - (vec->count + 1)) * sizeof(BLOP_VECTOR_DATA_TYPE));
440
441 free(vec->data);
442 vec->data = data;
443 vec->capacity = capacity;
444 vec->count++;
445 return;
446 }
447
448 memmove(&vec->data[idx + 1], &vec->data[idx], (vec->count - idx) * sizeof(BLOP_VECTOR_DATA_TYPE));
449 vec->data[idx] = value;
450 vec->count++;
451}
452void blopfn_vector_push_back(blopt_vector vec, BLOP_VECTOR_DATA_TYPE value) {
453 BLOP_ASSERT_PTR_VOID(vec);
454
455 if (vec->count == vec->capacity) {
456 vec->capacity *= BLOP_VECTOR_SCALATOR;
457 BLOP_VECTOR_DATA_TYPE* data = BLOP_MALLOC(BLOP_VECTOR_DATA_TYPE, vec->capacity);
458 if (!data) {
459 BLOP_ERROR_MESSAGE("Failed to allocate memory for vector data");
460 BLOP_ABORT();
461 return;
462 }
463
464 memcpy(vec->data, data, vec->count * sizeof(BLOP_VECTOR_DATA_TYPE));
465 free(vec->data);
466 vec->data = data;
467 }
468
469 vec->data[vec->count] = value;
470 vec->count++;
471}
472void blopfn_vector_push_front(blopt_vector vec, BLOP_VECTOR_DATA_TYPE value) {
473 BLOP_ASSERT_PTR_VOID(vec);
474
475 if (vec->count == vec->capacity) {
476 size_t capacity = vec->capacity * BLOP_VECTOR_SCALATOR;
477 BLOP_VECTOR_DATA_TYPE* data = (BLOP_VECTOR_DATA_TYPE*)calloc(capacity, sizeof(BLOP_VECTOR_DATA_TYPE));
478 if (!data) {
479 BLOP_ERROR_MESSAGE("Failed to allocate memory for vector data (returning without effect)");
480 BLOP_ABORT();
481 return;
482 }
483
484 memcpy(&data[1], vec->data, vec->count * sizeof(BLOP_VECTOR_DATA_TYPE));
485 data[0] = value;
486
487 free(vec->data);
488 vec->data = data;
489 vec->capacity = capacity;
490 vec->count++;
491 return;
492 }
493
494 if (vec->count > 0) {
495 memmove(&vec->data[1], &vec->data[0], vec->count * sizeof(BLOP_VECTOR_DATA_TYPE));
496 }
497 vec->data[0] = value;
498 vec->count++;
499}

520void blopfn_vector_erase(blopt_vector vec, size_t idx) {
521 BLOP_ASSERT_PTR_VOID(vec);
522
523 BLOP_ASSERT_BONDS_VOID(idx, vec->count);
524
525 if (idx == 0) {
526 blopfn_vector_pop_front(vec);
527 return;
528 }
529
530 if (idx == vec->count - 1) {
531 blopfn_vector_pop_back(vec);
532 return;
533 }
534
535 #ifdef BLOP_VECTOR_DEALLOCATE_DATA
536 BLOP_VECTOR_DEALLOCATE_DATA(vec->data[idx]);
537 #endif /* BLOP_VECTOR_DEALLOCATE_DATA */
538
539 memmove(&vec->data[idx], &vec->data[idx + 1], ((vec->count - idx) - 1) * sizeof(BLOP_VECTOR_DATA_TYPE));
540 vec->count--;
541
542 blopfn_vector_shrink(vec);
543}

580void blopfn_vector_resize(blopt_vector vec, size_t count) {
581 BLOP_ASSERT_PTR_VOID(vec);
582
583 if (count == vec->count) {
584 return;
585 }
586
587 if (count < vec->count) {
588 #ifdef BLOP_VECTOR_DEALLOCATE_DATA
589 for (size_t i = count; i < vec->count; i++) {
590 BLOP_VECTOR_DEALLOCATE_DATA(vec->data[i]);
591 }
592 #endif /* BLOP_VECTOR_DEALLOCATE_DATA */
593 vec->count = count;
594 blopfn_vector_shrink(vec);
595 return;
596 }
597
598 if (count > vec->capacity) {
599 vec->capacity = count;
600 BLOP_VECTOR_DATA_TYPE* data = BLOP_MALLOC(BLOP_VECTOR_DATA_TYPE, vec->capacity);
601 if (!data) {
602 BLOP_ERROR_MESSAGE("Failed to allocate memory for vector->data");
603 BLOP_ABORT();
604 return;
605 }
606
607 memcpy(data, vec->data, vec->count * sizeof(BLOP_VECTOR_DATA_TYPE));
608 free(vec->data);
609 vec->data = data;
610 }
611
612 memset(&vec->data[vec->count], 0, (count - vec->count) * sizeof(BLOP_VECTOR_DATA_TYPE));
613 vec->count = count;
614}
615void blopfn_vector_shrink(blopt_vector vec) {
616 BLOP_ASSERT_PTR_VOID(vec);
617
618 if (vec->count < vec->capacity / (BLOP_VECTOR_SCALATOR * 2) && vec->count > BLOP_VECTOR_INITIAL_COUNT) {
619 vec->capacity = vec->count * BLOP_VECTOR_SCALATOR;
620 vec->capacity = (vec->capacity < BLOP_VECTOR_INITIAL_COUNT) ? BLOP_VECTOR_INITIAL_COUNT : vec->capacity;
621 BLOP_VECTOR_DATA_TYPE* data = BLOP_MALLOC(BLOP_VECTOR_DATA_TYPE, vec->capacity);
622 if (!data) {
623 BLOP_ERROR_MESSAGE("Failed to allocate memory for vector->data");
624 BLOP_ABORT();
625 return;
626 }
627
628 memcpy(data, vec->data, vec->count * sizeof(BLOP_VECTOR_DATA_TYPE));
629 free(vec->data);
630 vec->data = data;
631 }
632}
