317blopt_node blopfn_list_get_node(blopt_list list, size_t idx) {
318 BLOP_ASSERT_PTR(list, NULL);
319
320 BLOP_ASSERT_BONDS(idx, list->size, NULL);
321
322 blopt_node current = NULL;
323 if (idx < list->size / 2) {
324 current = list->front;
325 for (size_t i = 0; i < idx; i++) {
326 current = current->next;
327 }
328 } else {
329 current = list->back;
330 for (size_t i = list->size - 1; i > idx; i--) {
331 current = current->prev;
332 }
333 }
334 return current;
335}

337void blopfn_list_clear(blopt_list list, int deallocate) {
338 BLOP_ASSERT_PTR_VOID(list);
339
340 if (list->size == 0) {
341 return;
342 }
343
344 blopt_node current = list->front;
345 blopt_node next = NULL;
346 while (current) {
347 next = current->next;
348 current->list = NULL;
349 current->next = NULL;
350 current->prev = NULL;
351 if (deallocate) {
352 blopfn_node_destroy(current);
353 }
354 current = next;
355 }
356
357 list->size = 0;
358 list->front = NULL;
359 list->back = NULL;
360}
361void blopfn_list_erase(blopt_list list, blopt_node node, int deallocate) {
362 BLOP_ASSERT_PTR_VOID(list);
363 BLOP_ASSERT_PTR_VOID(node);
364
365 BLOP_ASSERT_VOID(node->list == list, "The node does not belong to this list");
366
367 if (node->prev) {
368 node->prev->next = node->next;
369 } else {
370 list->front = node->next;
371 }
372
373 if (node->next) {
374 node->next->prev = node->prev;
375 } else {
376 list->back = node->prev;
377 }
378
379 list->size--;
380
381 node->list = NULL;
382 node->next = NULL;
383 node->prev = NULL;
384 if (deallocate) {
385 blopfn_node_destroy(node);
386 }

389 BLOP_ASSERT_PTR_VOID(list);
390
391 if (list->size == 0) {

393 return;

397
