684blopt_node blopfn_list_get_node(blopt_list list, size_t idx) {
685 BLOP_ASSERT_PTR(list, NULL);
686
687 BLOP_ASSERT_BONDS(idx, list->size, NULL);
688
689 blopt_node current = NULL;
690 if (idx < list->size / 2) {
691 current = list->front;
692 for (size_t i = 0; i < idx; i++) {
693 current = current->next;
694 }
695 } else {
696 current = list->back;
697 for (size_t i = list->size - 1; i > idx; i--) {
698 current = current->prev;
699 }
700 }
701 return current;
702}

704void blopfn_list_clear(blopt_list list, int deallocate) {
705 BLOP_ASSERT_PTR_VOID(list);
706
707 if (list->size == 0) {
708 return;
709 }
710
711 blopt_node current = list->front;
712 blopt_node next = NULL;
713 while (current) {
714 next = current->next;
715 current->list = NULL;
716 current->next = NULL;
717 current->prev = NULL;
718 if (deallocate) {
719 blopfn_node_destroy(current);
720 }
721 current = next;
722 }
723
724 list->size = 0;
725 list->front = NULL;
726 list->back = NULL;
727}

729void blopfn_list_erase(blopt_list list, blopt_node node, int deallocate) {
730 BLOP_ASSERT_PTR_VOID(list);
731 BLOP_ASSERT_PTR_VOID(node);
732
733 BLOP_ASSERT_VOID(node->list == list, "The node does not belong to this list");
734
735 if (node->prev) {
736 node->prev->next = node->next;
737 } else {
738 list->front = node->next;
739 }
740
741 if (node->next) {
742 node->next->prev = node->prev;
743 } else {
744 list->back = node->prev;
745 }
746
747 list->size--;
748
749 node->list = NULL;
750 node->next = NULL;
751 node->prev = NULL;
752 if (deallocate) {
753 blopfn_node_destroy(node);
754 }
755}

758 BLOP_ASSERT_PTR_VOID(list);
759
760 if (list->size == 0) {
761 BLOP_EMPTY_POPPING("To enable empty popping and avoid this error define BLOP_VALID_EMPTY_POPPING");
762 return;

765 blopt_node back = list->back;
766

768 list->back->prev->next = NULL;

771 list->front = NULL;

813void blopfn_list_push_back(blopt_list list, blopt_node node) {
814 BLOP_ASSERT_PTR_VOID(list);
815 BLOP_ASSERT_PTR_VOID(node);
816
817 BLOP_ASSERT_VOID(node->list == NULL, "The node already belongs to a list, duplicate the node to obtain same data without belonging to a list or erase it from the belonging list.");
818
819 node->list = list;
820 node->next = NULL;
821 node->prev = list->back;
822
823 if (list->back) {
824 list->back->next = node;
825 list->back = node;
826 } else {
827 list->front = node;
828 list->back = node;
829 }
830
831 list->size++;
832}
