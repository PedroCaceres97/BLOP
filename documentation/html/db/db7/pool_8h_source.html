105void blopfn_pool_free(blopt_pool pool, void* ptr) {
106 BLOP_ASSERT_PTR_VOID(pool);
107 BLOP_ASSERT_PTR_VOID(ptr);
108
109 blopt_ptrhdr hdr = (blopt_ptrhdr)BLOP_PSUB(ptr, sizeof(struct blops_ptrhdr));
110 BLOP_ASSERT_VOID(hdr->pool == pool, "The pointer does not belong to this pool");
111
112 blopfn_pool_list_erase(&pool->ptrs, &hdr->node, false);
113 pool->bytes -= hdr->size;
114 free(hdr);
115}
116void blopfn_pool_free_all(blopt_pool pool) {
117 BLOP_ASSERT_PTR_VOID(pool);
118
119 blopt_pool_list_node current = blopfn_pool_list_get_front(&pool->ptrs);
120 while (current != NULL) {
121 blopfn_pool_list_pop_front(&pool->ptrs, false);
122 blopt_ptrhdr hdr = (blopt_ptrhdr)current->data;
123 pool->bytes -= hdr->size;
124 free(hdr);
125 current = blopfn_pool_list_get_front(&pool->ptrs);
126 }

133 blopt_ptrhdr hdr = BLOP_ALLOC(struct blops_ptrhdr, size + sizeof(struct blops_ptrhdr));

175void* blopfn_pool_realloc(blopt_pool pool, void* ptr, size_t size) {
176 BLOP_ASSERT_PTR(pool, NULL);
177 BLOP_ASSERT_PTR(ptr, NULL);
178
179 BLOP_ASSERT(size != 0, NULL, "Requested allocation size is 0");
180
181 blopt_ptrhdr srchdr = (blopt_ptrhdr)BLOP_PSUB(ptr, sizeof(struct blops_ptrhdr));
182 BLOP_ASSERT(srchdr->pool != pool, NULL, "The pointer does not belong to this pool");
183
184 blopt_ptrhdr newhdr = BLOP_ALLOC(struct blops_ptrhdr, size + sizeof(struct blops_ptrhdr));
185 if (!newhdr) {
186 BLOP_ERROR_MESSAGE("Failed to allocate memory for pointer header");
187 BLOP_ABORT();
188 return NULL;
189 }
190
191 newhdr->size = size;
192 newhdr->pool = pool;
193 newhdr->node.data = newhdr;
194
195 blopfn_pool_list_insert_next(&pool->ptrs, &srchdr->node, &newhdr->node);
196
197 pool->bytes += size;
198 void* newptr = BLOP_PADD(newhdr, sizeof(struct blops_ptrhdr));
199 memcpy(newptr, ptr, BLOP_MIN(size, srchdr->size));
200
201 if (size > srchdr->size) {
202 memset(BLOP_PADD(newptr, srchdr->size), 0, size - srchdr->size);
203 }
204
205 blopfn_pool_list_erase(&pool->ptrs, &srchdr->node, false);
206 pool->bytes -= srchdr->size;
207 free(srchdr);
208
209 return newptr;
210}
